/** O(1) removal: swap target with last element, then pop. Reverse-iteration safe. */
function swapRemove<T>(arr: T[], index: number): void {
  const last = arr.length - 1;
  if (index !== last) {
    arr[index] = arr[last];
  }
  arr.pop();
}
```
